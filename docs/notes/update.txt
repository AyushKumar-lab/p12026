from selenium import webdriver
from selenium.webdriver.common.by import By
import time
import json

def get_popular_times(place_name):
   """
   FREE method to get Google Maps popular times
   No API key needed. Runs on your computer.
   """
   # Open Chrome browser (free)
   driver = webdriver.Chrome()
   
   # Search Google Maps
   search_url = f"https://www.google.com/maps/search/{place_name.replace(' ', '+')}"
   driver.get(search_url)
   time.sleep(5)  # Wait for page to load
   
   # Click first result
   try:
       first_result = driver.find_element(By.CLASS_NAME, "hfpxzc")
       first_result.click()
       time.sleep(3)
   except:
       pass
   
   # Look for popular times chart
   popular_times = {}
   try:
       # Find the histogram bars
       bars = driver.find_elements(By.CSS_SELECTOR, "[aria-label*='busy']")
       for bar in bars:
           label = bar.get_attribute('aria-label')
           # Parse: "Tuesday 12 PM, 75% busy"
           popular_times['raw_data'] = label
   except:
       popular_times['error'] = 'No popular times data found'
   
   driver.quit()
   return popular_times

# Example usage
if __name__ == "__main__":
   result = get_popular_times("Starbucks, Manhattan, New York")
   print(json.dumps(result, indent=2))

import osmnx as ox
import pandas as pd

def get_competitors(location, business_type, radius_meters=1000):
   """
   FREE method to get all competitors from OpenStreetMap
   No API key. No account. Unlimited usage.
   """
   # Download OSM data for the area
   tags = {'amenity': business_type}
   
   # Get geometries from place name
   gdf = ox.geometries_from_place(location, tags)
   
   # Convert to simple dataframe
   competitors = []
   for idx, row in gdf.iterrows():
       competitor = {
           'name': row.get('name', 'Unknown'),
           'lat': row.geometry.centroid.y,
           'lon': row.geometry.centroid.x,
           'type': row.get('amenity', business_type),
           'address': row.get('addr:street', 'No address')
       }
       competitors.append(competitor)
   
   return pd.DataFrame(competitors)

def get_walkability(location):
   """
   FREE walkability score from street network
   """
   # Download street network
   G = ox.graph_from_place(location, network_type='walk')
   
   # Calculate stats
   stats = ox.basic_stats(G)
   
   return {
       'street_density': stats['street_density_km'],
       'intersection_count': stats['intersection_count'],
       'average_street_length': stats['street_length_avg']
   }

# Example usage
if __name__ == "__main__":

   print(cafes.head())
   
   # Get walkability
   walk = get_walkability(location)
   print(f"Walkability score: {walk}")

import pandas as pd
import requests

def download_census_data(state_fips='36', county_fips='061'):
   """
   FREE download of Census demographic data
   No account needed. Direct CSV download.
   """
   # Build URL for ACS 5-year estimates
   # This gets income, population, age for all census tracts
   
   base_url = "https://api.census.gov/data/2022/acs/acs5"
   
   # Variables we want (FREE):
   # B01003_001E = Total population
   # B19013_001E = Median household income
   # B25064_001E = Median gross rent
   # B08303_001E = Mean travel time to work
   
   variables = "B01003_001E,B19013_001E,B25064_001E,B08303_001E"
   
   # Construct URL
   url = f"{base_url}?get={variables}&for=tract:*&in=state:{state_fips}&in=county:{county_fips}"
   
   # Download (FREE)
   response = requests.get(url)
   data = response.json()
   
   # Convert to dataframe
   columns = data[0]
   rows = data[1:]
   df = pd.DataFrame(rows, columns=columns)
   
   # Rename columns for readability
   df = df.rename(columns={
       'B01003_001E': 'population',
       'B19013_001E': 'median_income',
       'B25064_001E': 'median_rent',
       'B08303_001E': 'avg_commute_time',
       'tract': 'census_tract'
   })
   
   return df

# Alternative: Direct download from Census FTP (no API)
def download_tiger_shapefile(state='ny', county='061'):
   """
   Download TIGER/Line shapefiles directly
   URL pattern is public and free
   """
   url = f"https://www2.census.gov/geo/tiger/TIGER2022/TRACT/tl_2022_{state}_tract.zip"
   
   # Download file
   r = requests.get(url)
   with open(f'census_tracts_{state}.zip', 'wb') as f:
       f.write(r.content)
   
   print(f"Downloaded: census_tracts_{state}.zip")
   return f'census_tracts_{state}.zip'

import pandas as pd
import numpy as np

def calculate_location_score(location_data):
   """
   FREE algorithm to calculate business location score
   Combines multiple free data sources
   """
   score = 0
   factors = {}
   
   # Factor 1: Population Density (from Census)
   pop_density = float(location_data.get('population', 0)) / float(location_data.get('area_sqkm', 1))
   factors['population_score'] = min(pop_density / 10000, 10)  # Cap at 10
   score += factors['population_score']
   
   # Factor 2: Income Level (from Census)
   income = float(location_data.get('median_income', 50000))
   factors['income_score'] = min((income - 30000) / 5000, 10)  # Higher income = better
   score += factors['income_score']
   
   # Factor 3: Competition Density (from OSM)
   competitor_count = int(location_data.get('competitor_count', 0))
   # Inverse score - too many competitors is bad
   if competitor_count < 3:
       factors['competition_score'] = 10  # Low competition
   elif competitor_count < 10:
       factors['competition_score'] = 5   # Medium
   else:
       factors['competition_score'] = 2   # High competition
   score += factors['competition_score']
   
   # Factor 4: Transit Access (from OSM)
   transit_stops = int(location_data.get('transit_stops', 0))
   factors['transit_score'] = min(transit_stops * 2, 10)
   score += factors['transit_score']
   
   # Factor 5: Walkability (from OSM street network)
   intersection_density = float(location_data.get('intersection_density', 0))
   factors['walkability_score'] = min(intersection_density / 10, 10)
   score += factors['walkability_score']
   
   # Calculate final score (out of 50, convert to 10)
   final_score = (score / 50) * 10
   
   return {
       'total_score': round(final_score, 1),
       'factors': factors,
       'rating': 'Excellent' if final_score > 8 else 'Good' if final_score > 6 else 'Average'
   }

import folium

def create_free_map(lat, lon, competitors_df, filename='location_map.html'):
   """
   Create interactive map using FREE Folium library
   No API key needed. No account needed.
   """
   # Create base map
   m = folium.Map(location=[lat, lon], zoom_start=15)
   
   # Add marker for target location
   folium.Marker(
       [lat, lon],
       popup='YOUR BUSINESS LOCATION',
       icon=folium.Icon(color='red', icon='star')
   ).add_to(m)
   
   # Add competitor markers
   for idx, row in competitors_df.iterrows():
       folium.CircleMarker(
           location=[row['lat'], row['lon']],
           radius=6,
           popup=row['name'],
           color='blue',
           fill=True,
           fill_color='blue'
       ).add_to(m)
   
   # Add heatmap layer (competitor density)
   heat_data = [[row['lat'], row['lon']] for idx, row in competitors_df.iterrows()]
   from folium.plugins import HeatMap
   HeatMap(heat_data).add_to(m)
   
   # Save to HTML (FREE)
   m.save(filename)
   print(f"Map saved: {filename}")
   print("Open this file in any web browser")
   
   return filename
